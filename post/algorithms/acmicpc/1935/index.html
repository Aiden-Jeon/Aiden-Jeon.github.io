import sys


def calculator(after, before, cur):
    if cur == "+":
        return before + after
    elif cur == "-":
        return before - after
    elif cur == "*":
        return before * after
    elif cur == "/":
        return before / after


N = int(sys.stdin.readline())
math = sys.stdin.readline().strip()

alphabet = ord("A")
replace = {}
for _ in range(N):
    n = int(sys.stdin.readline())
    replace[chr(alphabet)] = n
    alphabet += 1

stacks = []
for cur in math:
    if ord("A") <= ord(cur) <= ord("Z"):
        stacks.append(replace[cur])
    else:
        after = stacks.pop()
        before = stacks.pop()
        stacks.append(calculator(after, before, cur))

print("{:.2f}".format(stacks[0]))
